import re
import nltk


def getTheRestAsString(text):
    match=re.search(theRestPattern, text)
    return  match.group()
    
def getTypes(text):
    tokens = nltk.word_tokenize(text)
    tagged = nltk.pos_tag(tokens)
    nns = [x[0] for x in tagged if (x[1]=='NN' or x[1]=='NNS')]
    result = ""
    for n in nns:
        result += (n + ", ")
    return result
